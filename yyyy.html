#include <iostream>
#include <string>
#include <cctype>

using namespace std;

// Function to preprocess the plaintext
string preprocessText(const string& text) {
    string processedText = "";
    for (char c : text) {
        if (isalpha(c)) {
            processedText += toupper(c);
        }
    }
    return processedText;
}

// Function to generate the Playfair key matrix
void generateKeyMatrix(const string& key, char keyMatrix[5][5]) {
    string uniqueChars = "";
    for (char c : key) {
        if (isalpha(c) && uniqueChars.find(toupper(c)) == string::npos) {
            uniqueChars += toupper(c);
        }
    }

    // Fill the key matrix with unique characters from the key
    int i = 0;
    int j = 0;
    for (char c : uniqueChars) {
        keyMatrix[i][j] = c;
        j++;
        if (j == 5) {
            i++;
            j = 0;
        }
    }

    // Fill the remaining cells with the remaining alphabets
    char currentChar = 'A';
    while (i < 5) {
        while (j < 5) {
            if (uniqueChars.find(currentChar) == string::npos && currentChar != 'J') {
                keyMatrix[i][j] = currentChar;
                j++;
            }
            if (j == 5) {
                i++;
                j = 0;
            }
            currentChar++;
        }
    }
}

// Function to find the positions of two characters in the key matrix
void findPositions(const char keyMatrix[5][5], char a, char b, int& rowA, int& colA, int& rowB, int& colB) {
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 5; j++) {
            if (keyMatrix[i][j] == a) {
                rowA = i;
                colA = j;
            } else if (keyMatrix[i][j] == b) {
                rowB = i;
                colB = j;
            }
        }
    }
}

// Function to encrypt a digraph using the Playfair cipher
string encryptDigraph(const char keyMatrix[5][5], char a, char b) {
    int rowA, colA, rowB, colB;
    findPositions(keyMatrix, a, b, rowA, colA, rowB, colB);
    string encrypted = "";

    // If both characters are in the same row
    if (rowA == rowB) {
        encrypted += keyMatrix[rowA][(colA + 1) % 5];
        encrypted += keyMatrix[rowB][(colB + 1) % 5];
    }
    // If both characters are in the same column
    else if (colA == colB) {
        encrypted += keyMatrix[(rowA + 1) % 5][colA];
        encrypted += keyMatrix[(rowB + 1) % 5][colB];
    }
    // If characters form a rectangle
    else {
        encrypted += keyMatrix[rowA][colB];
        encrypted += keyMatrix[rowB][colA];
    }

    return encrypted;
}

// Function to encrypt plaintext using the Playfair cipher
string encryptPlayfair(const string& plaintext, const string& key) {
    char keyMatrix[5][5];
    generateKeyMatrix(key, keyMatrix);

    string processedText = preprocessText(plaintext);
    string encryptedText = "";

    // Encrypt digraphs in the processed plaintext
    for (int i = 0; i < processedText.length(); i += 2) {
        encryptedText += encryptDigraph(keyMatrix, processedText[i], processedText[i + 1]);
    }

    return encryptedText;
}

int main() {
    string plaintext = "HELLO";
    string key = "WORLD";
    cout << "Plaintext: " << plaintext << endl;
    cout << "Key: " << key << endl;

    string encryptedText = encryptPlayfair(plaintext, key);
    cout << "Encrypted text: " << encryptedText << endl;

    return 0;
}
